def extract_week_and_year(page):
    """
    Busca en el texto del PDF:
    "Vigilancia Epidemiol√≥gica Semana 9, 2025"
    
    Regex: r"Semana\s+(\d+),\s*(\d{4})"
    Grupos: 
      - \d+ : N√∫mero de semana
      - \d{4} : A√±o (4 d√≠gitos)
    """
    text = page.extract_text()

    return None, None
